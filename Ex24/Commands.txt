Commands Used

Create EKS Cluster
eksctl create cluster ^
--name ex24-cluster ^
--region us-east-1 ^
--nodegroup-name workers ^
--node-type t3.small ^
--nodes 2

Verify Cluster
kubectl get nodes

Create DynamoDB Table
aws dynamodb create-table ^
--table-name customer-data ^
--attribute-definitions AttributeName=CustomerId,AttributeType=S ^
--key-schema AttributeName=CustomerId,KeyType=HASH ^
--billing-mode PAY_PER_REQUEST ^
--region us-east-1

Associate OIDC
eksctl utils associate-iam-oidc-provider ^
--cluster ex24-cluster ^
--region us-east-1 ^
--approve

Create IAM Policy
aws iam create-policy ^
--policy-name DynamoDBCustomerPolicy ^
--policy-document file://dynamodb-policy.json

Create Service Account
eksctl create iamserviceaccount ^
--cluster ex24-cluster ^
--namespace default ^
--name customer-sa ^
--attach-policy-arn arn:aws:iam::810648236794:policy/DynamoDBCustomerPolicy ^
--approve ^
--override-existing-serviceaccounts ^
--region us-east-1

Build Application
mvn clean package

Build Docker Image
docker build -t customer-service:v4 .

Create ECR Repository
aws ecr create-repository ^
--repository-name customer-service ^
--region us-east-1

Login to ECR
aws ecr get-login-password --region us-east-1 | docker login --username AWS --password-stdin 810648236794.dkr.ecr.us-east-1.amazonaws.com

Tag Image
docker tag customer-service:v4 810648236794.dkr.ecr.us-east-1.amazonaws.com/customer-service:latest

Push Image
docker push 810648236794.dkr.ecr.us-east-1.amazonaws.com/customer-service:latest

Deploy Application
kubectl apply -f deployment.yaml
kubectl apply -f service.yaml

Verify Deployment
kubectl get pods
kubectl get svc

Restart Deployment
kubectl rollout restart deployment customer-service

API Testing
Create Customer
curl -X POST http://<ELB>/customer ^
-H "Content-Type: application/json" ^
-d "{\"customerId\":\"101\",\"name\":\"Lalith\"}"

Response:
Customer Created

Read Customer
curl http://<ELB>/customer/101

Response:
{
  "customerId":"101",
  "name":"Lalith"
}

Update Customer
curl -X PUT http://<ELB>/customer/101 ^
-H "Content-Type: application/json" ^
-d "{\"name\":\"Lalith Kumar\"}"

Response:
Customer Updated
