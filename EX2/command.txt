# Verify cluster
kubectl get nodes

# Enable OIDC provider
eksctl utils associate-iam-oidc-provider --cluster irsa-lab --region us-east-1 --approve

# Verify OIDC
aws eks describe-cluster --name irsa-lab --region us-east-1 --query "cluster.identity.oidc.issuer"

# Create DynamoDB table
aws dynamodb create-table --table-name customer-data --attribute-definitions AttributeName=customerId,AttributeType=S --key-schema AttributeName=customerId,KeyType=HASH --billing-mode PAY_PER_REQUEST --region us-east-1

# Create IAM policy
aws iam create-policy --policy-name DynamoDBIRSALabPolicy --policy-document file://dynamodb-policy.json

# Create namespace
kubectl create namespace irsa-demo

# Create IRSA service account
eksctl create iamserviceaccount customer-sa --cluster irsa-lab --namespace irsa-demo --attach-policy-arn arn:aws:iam::810648236794:policy/DynamoDBIRSALabPolicy --approve --override-existing-serviceaccounts

# Verify service account
kubectl get sa customer-sa -n irsa-demo -o yaml

# Deploy bad pod
kubectl apply -f bad-pod.yaml

# Verify pod
kubectl get pods -n irsa-demo

# Enter pod
kubectl exec -it bad-pod -n irsa-demo -- sh

# Check AWS identity (shows node role)
aws sts get-caller-identity

# Reproduce failure
aws dynamodb get-item --table-name customer-data --key '{"customerId":{"S":"1"}}' --region us-east-1

# Delete bad pod
kubectl delete pod bad-pod -n irsa-demo

# Deploy fixed pod
kubectl apply -f good-pod.yaml

# Verify fixed pod
kubectl get pods -n irsa-demo

# Enter fixed pod
kubectl exec -it good-pod -n irsa-demo -- sh

# Verify IRSA role
aws sts get-caller-identity

# Verify DynamoDB access
aws dynamodb get-item --table-name customer-data --key '{"customerId":{"S":"1"}}' --region us-east-1
