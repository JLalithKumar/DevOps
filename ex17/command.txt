Create EKS Cluster:

eksctl create cluster ^
--name ex17-irsa-cluster ^
--region us-east-1 ^
--nodes 2 ^
--node-type t3.small

Verify:

kubectl get nodes

Create DynamoDB Table:

aws dynamodb create-table ^
--table-name customer-data ^
--attribute-definitions AttributeName=CustomerId,AttributeType=S ^
--key-schema AttributeName=CustomerId,KeyType=HASH ^
--billing-mode PAY_PER_REQUEST ^
--region us-east-1

Verify:

aws dynamodb list-tables --region us-east-1

Associate OIDC Provider:

eksctl utils associate-iam-oidc-provider ^
--region us-east-1 ^
--cluster ex17-irsa-cluster ^
--approve

Create IAM Policy:

aws iam create-policy ^
--policy-name DynamoDBIRSAPolicy ^
--policy-document file://dynamodb-policy.json

Create IRSA Service Account:

eksctl create iamserviceaccount ^
--cluster ex17-irsa-cluster ^
--namespace default ^
--name dynamodb-sa ^
--attach-policy-arn arn:aws:iam::810648236794:policy/DynamoDBIRSAPolicy ^
--approve ^
--region us-east-1

Verify:

kubectl get sa
kubectl describe sa dynamodb-sa

Deploy Test Pod:

kubectl apply -f dynamodb-test-pod.yaml

Verify:

kubectl get pods

Enter Pod:

kubectl exec -it dynamodb-test -- sh

Verify IRSA:

aws sts get-caller-identity

Put Item:

aws dynamodb put-item \
--table-name customer-data \
--item '{"CustomerId":{"S":"101"},"Name":{"S":"Lalith"}}' \
--region us-east-1

Get Item:

aws dynamodb get-item \
--table-name customer-data \
--key '{"CustomerId":{"S":"101"}}' \
--region us-east-1

Update Item:

aws dynamodb update-item \
--table-name customer-data \
--key '{"CustomerId":{"S":"101"}}' \
--update-expression "SET #n = :newname" \
--expression-attribute-names '{"#n":"Name"}' \
--expression-attribute-values '{":newname":{"S":"Lalith Kumar"}}' \
--region us-east-1

Verify Update:

aws dynamodb get-item \
--table-name customer-data \
--key '{"CustomerId":{"S":"101"}}' \
--region us-east-1
